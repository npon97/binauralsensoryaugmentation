45 CNMEAParserData::ERROR_E CNMEASentenceRMC::ProcessSentence(char *pCmd, char *pData) {
46 
47  UNUSED_PARAM(pCmd);
48  char szField[c_nMaxField];
49 
50  // Time
51  if (GetField(pData, szField, 0, c_nMaxField) == CNMEAParserData::ERROR_OK) {
52  m_SentenceData.m_nHour = (szField[0] - '0') * 10 + (szField[1] - '0');
53  m_SentenceData.m_nMinute = (szField[2] - '0') * 10 + (szField[3] - '0');
54  m_SentenceData.m_nSecond = (szField[4] - '0') * 10 + (szField[5] - '0');
55  }
56 
57  // Status
58  if (GetField(pData, szField, 1, c_nMaxField) == CNMEAParserData::ERROR_OK) {
59  m_SentenceData.m_nStatus = (CNMEAParserData::RMC_STATUS_E)((char)szField[0]);
60  }
61  else {
62  m_SentenceData.m_nStatus = (CNMEAParserData::RMC_STATUS_VOID);
63  }
64 
65  //
66  // Latitude
67  //
68  if (GetField(pData, szField, 2, c_nMaxField) == CNMEAParserData::ERROR_OK)
69  {
70  m_SentenceData.m_dLatitude = atof((char *)szField + 2) / 60.0;
71  szField[2] = '\0';
72  m_SentenceData.m_dLatitude += atof((char *)szField);
73 
74  }
75  if (GetField(pData, szField, 3, c_nMaxField) == CNMEAParserData::ERROR_OK)
76  {
77  if (szField[0] == 'S')
78  {
79  m_SentenceData.m_dLatitude = -m_SentenceData.m_dLatitude;
80  }
81  }
82 
83  //
84  // Longitude
85  //
86  if (GetField(pData, szField, 4, c_nMaxField) == CNMEAParserData::ERROR_OK)
87  {
88  m_SentenceData.m_dLongitude = atof((char *)szField + 3) / 60.0;
89  szField[3] = '\0';
90  m_SentenceData.m_dLongitude += atof((char *)szField);
91  }
92  if (GetField(pData, szField, 5, c_nMaxField) == CNMEAParserData::ERROR_OK)
93  {
94  if (szField[0] == 'W')
95  {
96  m_SentenceData.m_dLongitude = -m_SentenceData.m_dLongitude;
97  }
98  }
99 
100  // Speed over ground knots
101  if (GetField(pData, szField, 6, c_nMaxField) == CNMEAParserData::ERROR_OK) {
102  m_SentenceData.m_dSpeedKnots = atol(szField);
103  }
104  else {
105  m_SentenceData.m_dSpeedKnots = 0.0;
106  }
107 
108  // Track Angle
109  if (GetField(pData, szField, 7, c_nMaxField) == CNMEAParserData::ERROR_OK) {
110  m_SentenceData.m_dTrackAngle = atof(szField);
111  }
112  else {
113  m_SentenceData.m_dTrackAngle = 0.0;
114  }
115 
116 
117  // Date
118  if (GetField(pData, szField, 8, c_nMaxField) == CNMEAParserData::ERROR_OK) {
119  // 23 03 94 Date - 23rd of March 1994
120  m_SentenceData.m_nDay = (szField[0] - '0') * 10 + (szField[1] - '0');
121  m_SentenceData.m_nMonth = (szField[2] - '0') * 10 + (szField[3] - '0');
122  m_SentenceData.m_nYear = (szField[4] - '0') * 10 + (szField[4] - '0');
123  m_SentenceData.m_nYear += 2000;
124  }
125  else {
126  m_SentenceData.m_nMonth = 0;
127  m_SentenceData.m_nDay = 0;
128  m_SentenceData.m_nYear = 0;
129  }
130 
131 
132  // Magnetic Variation
133  if (GetField(pData, szField, 9, c_nMaxField) == CNMEAParserData::ERROR_OK) {
134 
135  m_SentenceData.m_dMagneticVariation = atof(szField);
136 
137  if (GetField(pData, szField, 10, c_nMaxField) == CNMEAParserData::ERROR_OK) {
138  if(szField[0] == 'W') {
139  m_SentenceData.m_dMagneticVariation *= -1.0;
140  }
141  }
142  }
143  else {
144  m_SentenceData.m_dMagneticVariation = 0.0;
145  }
146 
147 
148  m_uRxCount++;
149 
150  return CNMEAParserData::ERROR_OK;
151 }
