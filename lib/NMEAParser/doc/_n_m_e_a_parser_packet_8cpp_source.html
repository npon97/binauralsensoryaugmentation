41 CNMEAParserData::ERROR_E CNMEAParserPacket::ProcessNMEABuffer(char * pData, size_t nBufferSize)
42 {
43  for (size_t i = 0; i < nBufferSize; i++) {
44  char cData = pData[i];
45  switch (m_nState)
46  {
47  ///////////////////////////////////////////////////////////////////////
48  // Search for start of message '$'
49  case PARSE_STATE_SOM:
50  //
51  // Data byte Start of Message?
52  //
53  if (cData == '$')
54  {
55  //
56  // Time tag this message
57  //
58  TimeTag();
59 
60  m_u8Checksum = 0; // reset checksum
61  m_nIndex = 0; // reset index
62  m_nState = PARSE_STATE_CMD;
63  }
64  break;
65 
66  ///////////////////////////////////////////////////////////////////////
67  // Retrieve command (NMEA Address)
68  case PARSE_STATE_CMD:
69  if (cData != ',' && cData != '*')
70  {
71  m_pCommand[m_nIndex++] = cData;
72  m_u8Checksum ^= cData;
73 
74  // Check for command overflow
75  if (m_nIndex >= CNMEAParserData::c_uMaxCmdLen)
76  {
77  OnError(CNMEAParserData::ERROR_CMD_BUFFER_OVERFLOW, m_pCommand);
78  m_nState = PARSE_STATE_SOM;
79  }
80  }
81  else
82  {
83  m_pCommand[m_nIndex] = '\0'; // terminate command
84  m_u8Checksum ^= cData;
85  m_nIndex = 0;
86  m_nState = PARSE_STATE_DATA; // goto get data state
87  }
88  break;
89 
90  ///////////////////////////////////////////////////////////////////////
91  // Store data and check for end of sentence or checksum flag
92  case PARSE_STATE_DATA:
93  if (cData == '*') // checksum flag?
94  {
95  m_pData[m_nIndex] = '\0';
96  m_nState = PARSE_STATE_CHECKSUM_1;
97  }
98  else // no checksum flag, store data
99  {
100  //
101  // Check for end of sentence with no checksum
102  //
103  if (cData == '\r')
104  {
105  m_pData[m_nIndex] = '\0';
106  ProcessRxCommand(m_pCommand, m_pData);
107  m_nState = PARSE_STATE_SOM;
108  return CNMEAParserData::ERROR_OK;
109  }
110 
111  //
112  // Store data and calculate checksum
113  //
114  m_u8Checksum ^= cData;
115  m_pData[m_nIndex] = cData;
116  // Increment and check next position in buffer
117  if (++m_nIndex >= CNMEAParserData::c_uMaxDataLen) // Check for buffer overflow
118  {
119  OnError(CNMEAParserData::ERROR_RX_BUFFER_OVERFLOW, m_pCommand);
120  m_nState = PARSE_STATE_SOM;
121  }
122  }
123  break;
124 
125  ///////////////////////////////////////////////////////////////////////
126  case PARSE_STATE_CHECKSUM_1:
127  if ((cData - '0') <= 9)
128  {
129  m_u8ReceivedChecksum = (cData - '0') << 4;
130  }
131  else
132  {
133  m_u8ReceivedChecksum = (cData - 'A' + 10) << 4;
134  }
135 
136  m_nState = PARSE_STATE_CHECKSUM_2;
137 
138  break;
139 
140  ///////////////////////////////////////////////////////////////////////
141  case PARSE_STATE_CHECKSUM_2:
142  if ((cData - '0') <= 9)
143  {
144  m_u8ReceivedChecksum |= (cData - '0');
145  }
146  else
147  {
148  m_u8ReceivedChecksum |= (cData - 'A' + 10);
149  }
150 
151  if (m_u8Checksum == m_u8ReceivedChecksum)
152  {
153  ProcessRxCommand(m_pCommand, m_pData);
154  }
155  // Checksum error
156  else {
157  OnError(CNMEAParserData::ERROR_CHECKSUM, m_pCommand);
158  }
159 
160  m_nState = PARSE_STATE_SOM;
161  break;
162 
163  ///////////////////////////////////////////////////////////////////////
164  default: m_nState = PARSE_STATE_SOM;
165  }
166  }
167  return CNMEAParserData::ERROR_OK;
168 }
