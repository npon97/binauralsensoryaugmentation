178 CNMEAParserData::ERROR_E CNMEAParser::ProcessRxCommand(char * pCmd, char * pData)
179 {
180 
181  //
182  // Grab the talker ID
183  //
184  uint16_t u16TalkerID = (uint16_t)((uint8_t)pCmd[0]) << 8;
185  u16TalkerID |= (uint16_t)((uint8_t)pCmd[1]);
186 
187  //
188  // Get the sentence ID, --XXX where XXX is the sentence ID
189  //
190  char *lpszSentenceID = &pCmd[2];
191 
192  printf("Cmd: %s, TalkerID:%c%c, Sentence ID: %s\n", pCmd, (u16TalkerID >> 8) & 0xFF, (u16TalkerID) & 0xFF, lpszSentenceID);
193 
194 
195  //-----------------------------------------------------------------------------
196  if (strcmp(pCmd, "GPGGA") == 0) {
197  DataAccessSemaphoreLock();

219  DataAccessSemaphoreUnlock();
220  }

223  m_GAGSA.ProcessSentence(pCmd, pData);
224  DataAccessSemaphoreUnlock();
