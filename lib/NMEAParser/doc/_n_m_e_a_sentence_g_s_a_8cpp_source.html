37 CNMEAParserData::ERROR_E CNMEASentenceGSA::ProcessSentence(char * pCmd, char * pData)
38 {
39  UNUSED_PARAM(pCmd);
40 
41  char szField[c_nMaxField];
42 
43  // Auto mode
44  if (GetField(pData, szField, 0, c_nMaxField) == CNMEAParserData::ERROR_OK) {
45  m_SentenceData.nAutoMode = (szField[0] == 'A') ? CNMEAParserData::ASAM_AUTO : CNMEAParserData::ASAM_MANUAL;
46  }
47  else {
48  m_SentenceData.nAutoMode = CNMEAParserData::ASAM_MANUAL;
49  }
50  // Fix mode
51  if (GetField(pData, szField, 1, c_nMaxField) == CNMEAParserData::ERROR_OK) {
52  m_SentenceData.nMode = (CNMEAParserData::ACTIVE_SAT_MODE_E)atoi(szField);
53  }
54  else {
55  m_SentenceData.nMode = CNMEAParserData::ASM_FIX_NOT_AVAILABLE;
56  }
57 
58  // Grab the satellite data
59  for (int i = 0; i < CNMEAParserData::c_nMaxGSASats; i++) {
60  if (GetField(pData, szField, 2 + i, c_nMaxField) == CNMEAParserData::ERROR_OK) {
61  m_SentenceData.pnPRN[i] = atoi(szField);
62  }
63  else {
64  m_SentenceData.pnPRN[i] = CNMEAParserData::c_nInvlidPRN;
65  }
66  }
67 
68  // PDOP
69  if (GetField(pData, szField, 14, c_nMaxField) == CNMEAParserData::ERROR_OK) {
70  m_SentenceData.dPDOP = atof(szField);
71  }
72  else {
73  m_SentenceData.dPDOP = 0.0;
74  }
75 
76  // HDOP
77  if (GetField(pData, szField, 15, c_nMaxField) == CNMEAParserData::ERROR_OK) {
78  m_SentenceData.dHDOP = atof(szField);
79  }
80  else {
81  m_SentenceData.dHDOP = 0.0;
82  }
83 
84  // VDOP
85  if (GetField(pData, szField, 16, c_nMaxField) == CNMEAParserData::ERROR_OK) {
86  m_SentenceData.dVDOP = atof(szField);
87  }
88  else {
89  m_SentenceData.dVDOP = 0.0;
90  }
91  return CNMEAParserData::ERROR_OK;
92 }
