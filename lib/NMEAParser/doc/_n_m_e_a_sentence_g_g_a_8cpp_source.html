38 CNMEAParserData::ERROR_E CNMEASentenceGGA::ProcessSentence(char * pCmd, char * pData)
39 {
40  UNUSED_PARAM(pCmd);
41  char szField[c_nMaxField];
42  char szTemp[c_nMaxField];
43 
44  if (GetField(pData, szField, 0, c_nMaxField) == CNMEAParserData::ERROR_OK) {
45  m_SentenceData.m_nHour = (szField[0] - '0') * 10 + (szField[1] - '0');
46  m_SentenceData.m_nMinute = (szField[2] - '0') * 10 + (szField[3] - '0');
47  m_SentenceData.m_nSecond = (szField[4] - '0') * 10 + (szField[5] - '0');
48  }
49 
50  //
51  // Latitude
52  //
53  if (GetField(pData, szField, 1, c_nMaxField) == CNMEAParserData::ERROR_OK)
54  {
55  m_SentenceData.m_dLatitude = atof((char *)szField + 2) / 60.0;
56  szField[2] = '\0';
57  m_SentenceData.m_dLatitude += atof((char *)szField);
58 
59  }
60  if (GetField(pData, szField, 2, c_nMaxField) == CNMEAParserData::ERROR_OK)
61  {
62  if (szField[0] == 'S')
63  {
64  m_SentenceData.m_dLatitude = -m_SentenceData.m_dLatitude;
65  }
66  }
67 
68  //
69  // Longitude
70  //
71  if (GetField(pData, szField, 3, c_nMaxField) == CNMEAParserData::ERROR_OK)
72  {
73  m_SentenceData.m_dLongitude = atof((char *)szField + 3) / 60.0;
74  szField[3] = '\0';
75  m_SentenceData.m_dLongitude += atof((char *)szField);
76  }
77  if (GetField(pData, szField, 4, c_nMaxField) == CNMEAParserData::ERROR_OK)
78  {
79  if (szField[0] == 'W')
80  {
81  m_SentenceData.m_dLongitude = -m_SentenceData.m_dLongitude;
82  }
83  }
84 
85  //
86  // GPS quality
87  //
88  if (GetField(pData, szField, 5, c_nMaxField) == CNMEAParserData::ERROR_OK)
89  {
90  m_SentenceData.m_nGPSQuality = (CNMEAParserData::GPS_QUALITY_E) ( (char)szField[0] - '0');
91  }
92 
93  //
94  // Satellites in use
95  //
96  if (GetField(pData, szField, 6, c_nMaxField) == CNMEAParserData::ERROR_OK)
97  {
98  szTemp[0] = szField[0];
99  szTemp[1] = szField[1];
100  szTemp[2] = '\0';
101  m_SentenceData.m_nSatsInView = atoi(szTemp);
102  }
103 
104  //
105  // HDOP
106  //
107  if (GetField(pData, szField, 7, c_nMaxField) == CNMEAParserData::ERROR_OK)
108  {
109  m_SentenceData.m_dHDOP = atof((char *)szField);
110  }
111 
112  //
113  // Altitude, Meters, above mean sea level
114  //
115  if (GetField(pData, szField, 8, c_nMaxField) == CNMEAParserData::ERROR_OK)
116  {
117  m_SentenceData.m_dAltitudeMSL = atof((char *)szField);
118  }
119 
120  //
121  // Geoidal separation, meters
122  //
123  if (GetField(pData, szField, 9, c_nMaxField) == CNMEAParserData::ERROR_OK)
124  {
125  m_SentenceData.m_dGeoidalSep = atof((char *)szField);
126  }
127 
128  //
129  // Differential age
130  //
131  if (GetField(pData, szField, 11, c_nMaxField) == CNMEAParserData::ERROR_OK)
132  {
133  m_SentenceData.m_dDifferentialAge = atof((char *)szField);
134  }
135 
136  //
137  // Differential ID
138  //
139  if (GetField(pData, szField, 13, c_nMaxField) == CNMEAParserData::ERROR_OK)
140  {
141  m_SentenceData.m_nDifferentialID = atoi((char *)szField);
142  }
143 
144  //
145  // Derive vertical speed (bonus)
146  //
147  int nSeconds = (int)m_SentenceData.m_nMinute * 60 + (int)m_SentenceData.m_nSecond;
148  if (nSeconds > m_nOldVSpeedSeconds)
149  {
150  double dDiff = (double)(m_nOldVSpeedSeconds - nSeconds);
151  double dVal = dDiff / 60.0;
152  if (dVal != 0.0)
153  {
154  m_SentenceData.m_dVertSpeed = (m_dOldVSpeedAlt - m_SentenceData.m_dAltitudeMSL) / dVal;
155  }
156  }
157  m_dOldVSpeedAlt = m_SentenceData.m_dAltitudeMSL;
158  m_nOldVSpeedSeconds = nSeconds;
159 
160  m_uRxCount++;
161 
162  return CNMEAParserData::ERROR_OK;
163 }
