39 CNMEAParserData::ERROR_E CNMEASentenceGSV::ProcessSentence(char * pCmd, char * pData)
40 {
41  UNUSED_PARAM(pCmd);
42  char szField[c_nMaxField];
43 
44  // Number of sentences
45  if (GetField(pData, szField, 0, c_nMaxField) == CNMEAParserData::ERROR_OK) {
46  m_SentenceData.nTotalNumberOfSentences = atoi(szField);
47  }
48 
49  // Number of sentences
50  if (GetField(pData, szField, 1, c_nMaxField) == CNMEAParserData::ERROR_OK) {
51  m_SentenceData.nSentenceNumber = atoi(szField);
52  }
53 
54  // Number of satellites in view
55  if (GetField(pData, szField, 2, c_nMaxField) == CNMEAParserData::ERROR_OK) {
56  m_SentenceData.nSatsInView = atoi(szField);
57  }
58 
59  for (int i = 0; i < 4; i++) {
60  // Calculate the index into the satellite data array base on the sentence number
61  int nIndex = (m_SentenceData.nSentenceNumber - 1) * 4 + i;
62  if ((nIndex + 4) >= CNMEAParserData::c_nMaxConstellation) {
63  return CNMEAParserData::ERROR_TOO_MANY_SATELLITES;
64  }
65 
66  // Get PRN
67  if (GetField(pData, szField, i*4 + 3, c_nMaxField) == CNMEAParserData::ERROR_OK) {
68  m_SentenceData.SatInfo[nIndex].nPRN = atoi(szField);
69  }
70  else {
71  m_SentenceData.SatInfo[nIndex].nPRN = CNMEAParserData::c_nInvlidPRN;
72  }
73  // Elevation
74  if (GetField(pData, szField, i * 4 + 4, c_nMaxField) == CNMEAParserData::ERROR_OK) {
75  m_SentenceData.SatInfo[nIndex].dElevation = atof(szField);
76  }
77  else {
78  m_SentenceData.SatInfo[nIndex].dElevation = 0.0;
79  }
80  // Azimuth
81  if (GetField(pData, szField, i * 4 + 5, c_nMaxField) == CNMEAParserData::ERROR_OK) {
82  m_SentenceData.SatInfo[nIndex].dAzimuth = atof(szField);
83  }
84  else {
85  m_SentenceData.SatInfo[nIndex].dAzimuth = atof(szField);
86  }
87  // Signal to noise
88  if (GetField(pData, szField, i * 4 + 6, c_nMaxField) == CNMEAParserData::ERROR_OK) {
89  m_SentenceData.SatInfo[nIndex].nSNR = atoi(szField);
90  }
91  else {
92  m_SentenceData.SatInfo[nIndex].nSNR = 0;
93  }
94  }
95 
96  // Check if this was the last sentence and clear the rest of the constellation data
97  if (m_SentenceData.nSentenceNumber == m_SentenceData.nTotalNumberOfSentences) {
98  for (int i = m_SentenceData.nTotalNumberOfSentences * 4; i < CNMEAParserData::c_nMaxConstellation; i++) {
99  m_SentenceData.SatInfo[i].nPRN = CNMEAParserData::c_nInvlidPRN;
100  m_SentenceData.SatInfo[i].dAzimuth = 0.0;
101  m_SentenceData.SatInfo[i].dElevation = 0.0;
102  m_SentenceData.SatInfo[i].nSNR = 0;
103  }
104  }
105 
106 
107  return CNMEAParserData::ERROR_OK;
108 }
